🐍 Python Script — Run Locally (pip install numpy) 
#!/usr/bin/env python3
"""
Solar ROI Calculator — 2026
Sources: NREL PVWatts, EnergySage 2025-26, local utility
"""

import numpy as np

# ─── USER INPUTS ──────────────────────────────────────────────────────────
SOLAR_KW        = 10.0    # System size in kW-DC
BATT_KWH        = 13.5    # Battery capacity kWh (0 = no battery)
MONTHLY_BILL    = 200.0   # Current avg monthly electric bill ($)
RATE_CKW        = 0.145   # Local utility rate $/kWh (adjust to match your bill)
DAILY_EV_MILES  = 30      # EV miles/day (0 = no EV)
EV_EFF          = 0.30    # kWh/mile (Model Y ~0.28-0.32)

SOLAR_PER_W     = 3.00    # $/W installed (panels + labor)
BATT_PER_KWH    = 900     # $/kWh installed (EG4/FranklinWH/Powerwall)
FIXED_FEES      = 3000    # Permits, interconnect, misc ($)

PEAK_SUN_HRS    = 4.6     # PVWatts: adjust for your location (US avg ~4.5)
SYSTEM_DERATE   = 0.85    # DC-to-AC, wiring, temp, soiling losses
DEGRADATION     = 0.005   # 0.5%/yr panel degradation
RATE_ESCALATION = 0.04    # 4%/yr rate escalation assumption
MAINTENANCE_YR  = 200     # Annual O&M ($)
BATT_REPLACE_YR = 12      # Battery replacement year
BATT_REPLACE_PCT= 0.60    # % of original battery cost to replace
YEARS           = 25      # System lifetime

BATT_RTE        = 0.95    # Battery round-trip efficiency

USE_LOAN        = False   # Set True for financing
LOAN_RATE       = 0.05    # 5% APR
LOAN_YEARS      = 10      # 10-year term

# ─── CALCULATIONS ─────────────────────────────────────────────────────────
# Gross cost
solar_cost  = SOLAR_KW * 1000 * SOLAR_PER_W
batt_cost   = BATT_KWH * BATT_PER_KWH
gross_cost  = solar_cost + batt_cost + FIXED_FEES
net_cost    = gross_cost

# Annual demand
home_kwh    = (MONTHLY_BILL / RATE_CKW) * 12
ev_kwh      = DAILY_EV_MILES * EV_EFF * 365
total_kwh   = home_kwh + ev_kwh

# Year 1 solar production
prod_yr1    = SOLAR_KW * PEAK_SUN_HRS * 365 * SYSTEM_DERATE

# Loan payment (if financing)
def pmt(r, n, pv):
    return pv * (r * (1+r)**n) / ((1+r)**n - 1)

annual_loan_pmt = pmt(LOAN_RATE/12, LOAN_YEARS*12, net_cost) * 12 if USE_LOAN else 0

cumulative_cf   = -net_cost if not USE_LOAN else 0
payback_year    = None
total_prod      = 0
total_savings_25= 0

print(f"\n{'='*60}")
print(f"  SOLAR ROI CALCULATOR — 2026")
print(f"{'='*60}")
print(f"  System:     {SOLAR_KW} kW solar + {BATT_KWH} kWh battery")
print(f"  Gross Cost: ${gross_cost:>10,.0f}")

print(f"  Net Cost:   ${net_cost:>10,.0f}")
print(f"  Annual Demand: {total_kwh:,.0f} kWh (home + EV)")
print(f"  Yr1 Production:{prod_yr1:,.0f} kWh")
print(f"\n  {'Yr':<4} {'Prod(kWh)':<12} {'Rate(¢)':<10} {'Savings($)':<12} {'Cum CF($)'}")
print(f"  {'-'*56}")

for yr in range(1, YEARS+1):
    production  = prod_yr1 * (1 - DEGRADATION) ** (yr-1)
    rate        = RATE_CKW * (1 + RATE_ESCALATION) ** (yr-1)
    offset      = min(production, total_kwh)
    export      = max(0, production - total_kwh)
    savings     = (offset + export) * rate - MAINTENANCE_YR
    savings    -= annual_loan_pmt  # subtract loan payment if financed

    # Battery replacement cost
    if yr == BATT_REPLACE_YR and BATT_KWH > 0:
        savings -= batt_cost * BATT_REPLACE_PCT

    cumulative_cf += savings
    total_prod    += production
    total_savings_25 += savings + (net_cost if yr==1 else 0)

    if cumulative_cf >= 0 and payback_year is None:
        payback_year = yr
        flag = " ← PAYBACK!"
    else:
        flag = ""

    print(f"  {yr:<4} {production:<12,.0f} {rate*100:<10.2f} {savings:<12,.0f} {cumulative_cf:>10,.0f}{flag}")

lcoe = net_cost / total_prod * 100 # cents per kWh
roi_ann = ((cumulative_cf + net_cost) / net_cost) ** (1/25) - 1

print(f"\n{'='*60}")
print(f"  PAYBACK PERIOD:    {payback_year} years")
print(f"  25-YEAR NET PROFIT:${cumulative_cf + net_cost:>10,.0f}")
print(f"  ANNUALIZED ROI:    {roi_ann*100:.1f}%")
print(f"  LCOE:              {lcoe:.1f} ¢/kWh")
print(f"{'='*60}\n")
